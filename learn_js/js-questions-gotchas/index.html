								function MemoizationFunc() {

									const memoize = (callback) => {
										const cache = {};

										return (...args) => {
											let jsonArgs = JSON.stringify(args);
											const  res = cache[jsonArgs] || callback(...args);
											cache[jsonArgs] = res;
											return res;
										}
									}

debounce function


								function debounceFunc() {
                  const debounce = (callback, delay) => {
                    let allowedToCall = true;

										return function(...args){
											if(!allowedToCall) return;

											callback(...args);
											allowedToCall = false;

											setTimeout(()=>{
												allowedToCall = true;
											},delay)
										}
                  };

                  function sayHi() {
                    console.log("hi from ", Array.from(arguments));
                  }

                  let f = debounce(sayHi, 1000);
                  f(1); // executing 
                  f(2); // ignored

                  setTimeout(() => f(3), 100); // ignored (100 ms has passed)
                  setTimeout(() => f(4), 1100); // executing
                  setTimeout(() => f(5), 1500); // ignored ( only 400 ms has passed from the last call )
                }
